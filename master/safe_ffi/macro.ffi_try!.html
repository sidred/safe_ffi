macro_rules! ffi_try {
    ($result:expr) => {
        match $result {
            Ok(value)  => value,
            Err(error) => {
                let decorator = ::std::iter::repeat('-').take(50).collect::<String>();
                println!("\n\n {}\n| {:?}\n {}\n\n", decorator, error, decorator);
                return error.into()
            },
        }
    }
}

This macro is intended to be used in all cases where we get Result and want to convert it to C friendly error code to inform about error conditions accross library boundaries

}
